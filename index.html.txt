        // Logic for Checklist Progress
        function toggleTopic(id) {
            if (checkedItems.has(id)) {
                checkedItems.delete(id);
            } else {
                checkedItems.add(id);
            }
            updateChecklistProgress();
        }

        function updateChecklistProgress() {
            const totalTopics = bioData.reduce((acc, mod) => acc + mod.topics.length, 0);
            const currentPercent = totalTopics > 0 ? Math.round((checkedItems.size / totalTopics) * 100) : 0;
            
            document.getElementById('checklist-percent').innerText = `${currentPercent}%`;
            document.getElementById('checklist-progress-bar').style.width = `${currentPercent}%`;
        }

        // Calculation and Generation of Dashboard Data
        function calculateAnalysis(event) {
            event.preventDefault();

            let totalWeightSum = 0;
            let weightedScoreSum = 0;

            const strengths = [];
            const weaknesses = [];
            const normal = [];

            bioData.forEach(mod => {
                const scoreValue = parseFloat(document.getElementById(`input-${mod.id}`).value) || 0;
                currentScores[mod.id] = scoreValue;

                const weightNum = parseFloat(mod.weight);
                totalWeightSum += weightNum;
                weightedScoreSum += (scoreValue * (weightNum / 100));

                if (scoreValue >= 75) {
                    strengths.push(mod);
                } else if (scoreValue < 60) {
                    weaknesses.push(mod);
                } else {
                    normal.push(mod);
                }
            });

            const overallScore = Math.round((weightedScoreSum / (totalWeightSum / 100)));

            // Update Dashboards Visuals
            switchTab('dashboard');
            renderDashboardResults(overallScore, strengths, weaknesses);
        }

        function renderDashboardResults(overallScore, strengths, weaknesses) {
            // Update Circle Progress
            document.getElementById('overall-score-txt').innerText = `${overallScore}%`;
            const circumference = 2 * Math.PI * 60; // 376.99
            const offset = circumference - (overallScore / 100) * circumference;
            document.getElementById('dashboard-radial-progress').style.strokeDashoffset = offset;

            // Status phrasing
            let statusText = "ต้องเร่งทบทวนเนื้อหาขนานใหญ่";
            if (overallScore >= 80) statusText = "เยี่ยมมาก! โอกาสผ่านเข้าค่ายสูงมาก";
            else if (overallScore >= 65) statusText = "พื้นฐานดี มีลุ้นสอบติด พยายามอุดจุดโหว่";

                            <div class="${barColor} h-full transition-all duration-1000" style="width: ${sc}%"></div>
                        </div>
                    </div>
                `;
            }).join('');

            // Lists rendering
            const sList = document.getElementById('strengths-list');

            } else {

                        <i data-lucide="check-circle-2" class="w-4 h-4 text-emerald-600 shrink-0"></i>

                `).join('');
            }

            const wList = document.getElementById('weaknesses-list');

            } else {
                wList.innerHTML = weaknesses.map(m => `
                    <li class="flex items-center gap-2 text-rose-900 font-medium">
                        <i data-lucide="x-circle" class="w-4 h-4 text-rose-600 shrink-0"></i>
                        ${m.title.split('. ')[1]} (${currentScores[m.id]}%)
                    </li>
                `).join('');
            }

            // AI/Smart Recommendations rendering
            const recContainer = document.getElementById('recommendations-container');
            if (weaknesses.length === 0) {
                recContainer.innerHTML = `
                    <div class="col-span-3 bg-gradient-to-r from-emerald-500/10 to-teal-500/10 border border-emerald-200/50 p-6 rounded-2xl flex items-center gap-4">
                        <div class="bg-emerald-500 text-white p-3 rounded-xl shadow-md"><i data-lucide="trophy" class="w-6 h-6"></i></div>
                        <div>
                            <h4 class="font-bold text-slate-900">สุดยอดมาก! คะแนนอยู่ในเกณฑ์มาตรฐานปลอดภัย</h4>
                            <p class="text-sm text-slate-600 mt-0.5">สิ่งที่คุณต้องทำตอนนี้คือจำลองข้อสอบแบบจับเวลาจริง 3 ชั่วโมง เพื่อรักษาเสถียรภาพและความเร็วในการคิด</p>
                        </div>
                    </div>
                `;
            } else {
                recContainer.innerHTML = weaknesses.map(m => `
                    <div class="bg-slate-50 border border-slate-200/80 p-5 rounded-2xl flex flex-col justify-between">
                        <div>
                            <div class="flex items-center gap-2 text-slate-900 font-bold text-sm mb-2">
                                <span class="w-2 h-2 bg-rose-500 rounded-full"></span>
                                ${m.title.split('. ')[1]}
                            </div>
                            <p class="text-xs text-slate-600 leading-relaxed">${adviceDatabase[m.id]}</p>

                            <span class="text-rose-600 font-semibold">ต้องเร่งแก้</span>
                        </div>
                    </div>
                `).join('');
            }

            // re-run lucide icons setup to render inside newly created markup elements
            lucide.createIcons();
